// 返回数字之和为n的数的下标
```javascript
var twoSum = function(nums,target) {
	// key: num ,value: index
	const numsMap = {}
	for(let i=0;i<nums.length;i++) {
		const cur = nums[i]
		if(numsMap[cur]!== undefined) {
			return [numsMap[cur],i]
		}
		else {
			const rest = target - cur
			numsMap[rest] = i
		}
	}
}
```
